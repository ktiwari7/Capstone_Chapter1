<script

}

<script type="text/javascript">
function create_map() {
var mapstraction = new mxn.Mapstraction('mymap', 'googlev3');
mapstraction.setCenterAndZoom(
new mxn.LatLonPoint(37.7740486,-122.4101883), 15);
}
</script>
</head>
<body onload="create_map()">
<div id="mymap"></div>
</body>
